458 public void OnAfterDie()
459 {
460 // Remove the modifier
461 RemoveModifierEffect();
462
463 // Reset the pieces into the scene
464 GameObject coins = GameObject.Find("Coins");
465 for (int i = 0; i < coins.transform.childCount; i++)
466 {
467 coins.transform.GetChild(i).gameObject.GetComponent<CoinColliderController>().ResetCoinPosition();
468 coins.transform.GetChild(i).gameObject.SetActive(true);
469 }
470
471 // Reset the modifier into the scene
472 GameObject modifiers = GameObject.Find("Modifiers");
473 for (int i = 0; i < modifiers.transform.childCount; i++)
474 modifiers.transform.GetChild(i).gameObject.SetActive(true);
475
476 // Reset the collected pieces of the player
477 CollectedCoins = 0;
478
479 // Updating the HUD
480 UpdateHudCoins();
481 UpdateHudAttempt();
482
483 // Respawn the player and reset velocity
484 player.WantsToJump = false;
485 player.Position = spawnPosition;
486 player.Velocity = new Vector2(0.0f, 0.0f);
487
488 // Reset positions of the sky parallax objects
489 GameObject.Find("Sky").GetComponent<ParallaxBackground>().ResetPos();
490 GameObject.Find("Cloud1").GetComponent<ParallaxBackground>().ResetPos();
491 GameObject.Find("Cloud2").GetComponent<ParallaxBackground>().ResetPos();
492 GameObject.Find("CloudsGroup").GetComponent<ParallaxBackground>().ResetPos();
493 }
