182 bool isOnGround()
183 {
184 float halfWidth = collider2d.size.x / 2.0f;
185 float rayDistance = collider2d.size.y / 2.0f + 0.05f;
186
187 // For each raycast (one on the left, one in the middle, one on the right)
188 for (int i = -1; i <= 1; i++)
189 {
190 Vector2 from = new Vector2(body.position.x + i * (halfWidth), body.position.y);
191 RaycastHit2D[] hits = Physics2D.RaycastAll(from, Vector2.down, rayDistance);
192
193 foreach (RaycastHit2D hit in hits)
194 {
195 if (hit.collider.name == "Terrain" ||
196 (hit.collider.name == "Nocive" && HasShield()))
197 {
198 return true;
199 }
200 }
201 }
202
203 return false;
204 }

301 private void RunUpdate()
302 {

305 velocity.y -= Gravity * Time.fixedDeltaTime;
306
307 body.MovePosition(body.position + velocity * Time.fixedDeltaTime);

309 if (isOnGround())

311 if (!wasOnGroundLastFrame)
312 {
313 wasOnGroundLastFrame = true;
314 audioSrc.PlayOneShot(ClipOnLandOnGround, 1.0f);
