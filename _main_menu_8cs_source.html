103 private void UpdateProgressBar()
104 {
105 float currentMaxDistancePercentage = (float) SaveManager.Instance.GameDatas.Levels["Level" + (SelectedLevelIndex + 1)].MaxDistance;
106
107 float leftWidthFull = ProgressBackgroundLeftPart.GetComponent<RectTransform>().sizeDelta.x;
108 float middleWidthFull = ProgressBackgroundMiddlePart.GetComponent<RectTransform>().sizeDelta.x;
109 float rightWidthFull = ProgressBackgroundRightPart.GetComponent<RectTransform>().sizeDelta.x;
110
111 float totalWidth = leftWidthFull + middleWidthFull + rightWidthFull;
112
113 Debug.Log("[MAIN MENU / PROGRESS BAR] progress percentage " + currentMaxDistancePercentage);
114 Debug.Log("[MAIN MENU / PROGRESS BAR] total length " + totalWidth);
115
116 float progressWidth = currentMaxDistancePercentage * totalWidth;
117
118 // left part
119 if (progressWidth == 0)
120 {
121 ProgressLeftPart.SetActive(false);
122 }
123 else
124 {
125 ProgressLeftPart.SetActive(true);
126 }
127
128 Debug.Log("[MAIN MENU / PROGRESS BAR] left part " + (ProgressLeftPart.activeSelf ? "ON" : "OFF"));
129
130 // middle part
131 float minTriggerPercentage = leftWidthFull / totalWidth;
132 float maxTriggerPercentage = (totalWidth - leftWidthFull) / totalWidth;
133
134 Debug.Log("[MAIN MENU / PROGRESS BAR] <" + minTriggerPercentage + " ; " + maxTriggerPercentage + ">");
135 if (currentMaxDistancePercentage >= minTriggerPercentage)
136 {
137 ProgressMiddlePart.SetActive(true);
138 float middleWidth = currentMaxDistancePercentage * middleWidthFull;
139 ProgressMiddlePart.GetComponent<RectTransform>().SetSizeWithCurrentAnchors(RectTransform.Axis.Horizontal, middleWidth);
140 }
141 else
142 {
143 ProgressMiddlePart.SetActive(false);
144 }
145
146 Debug.Log("[MAIN MENU / PROGRESS BAR] middle part " + (ProgressMiddlePart.activeSelf ? "ON" : "OFF"));
147
148 // right part
149 if (progressWidth == totalWidth)
150 {
151 ProgressRightPart.SetActive(true);
152 }
153 else
154 {
155 ProgressRightPart.SetActive(false);
156 }
157
158 Debug.Log("[MAIN MENU / PROGRESS BAR] right part " + (ProgressRightPart.activeSelf ? "ON" : "OFF"));
159 }
